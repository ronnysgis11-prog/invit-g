    <canvas id="matrix"></canvas>

    <script>
        const canvas = document.getElementById('matrix');
        const ctx = canvas.getContext('2d');
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;

        const letters = "ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789@#$%^&*()";
        const fontSize = 16;
        const columns = canvas.width / fontSize;
        const drops = Array(Math.floor(columns)).fill(1);

        function draw() {
            ctx.fillStyle = "rgba(0, 0, 0, 0.05)";
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            ctx.fillStyle = "#0F0";
            ctx.font = fontSize + "px monospace";

            drops.forEach((y, i) => {
                const text = letters[Math.floor(Math.random() * letters.length)];
                ctx.fillText(text, i * fontSize, y * fontSize);
                if (y * fontSize > canvas.height && Math.random() > 0.975) drops[i] = 0;
                drops[i]++;
            });
        }

        // Cambiar el mensaje progresivamente
        setTimeout(() => {
            document.getElementById('mensaje').innerHTML = "Hola [Su Nombre], me gusta mucho platicar contigo...";
        }, 3000);

        setTimeout(() => {
            document.getElementById('mensaje').innerHTML = "Me encantaría conocerte un poco más. <br><br><b>¿Te gustaría ir por un café algún día?</b>";
        }, 7000);

        setInterval(draw, 33);
    </script>
